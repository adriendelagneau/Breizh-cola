"use client"

import React, { useState, useEffect } from "react";
import { gsap } from "gsap";
import { Flip } from "gsap/Flip";
import Image from "next/image";

gsap.registerPlugin(Flip);

const FlipLayout = () => {
  const [layout, setLayout] = useState("stack");

  const flipLayout = async () => {
    // Get the current state of elements before DOM changes
    const state = Flip.getState(".circle", { props: "borderRadius" });

    // Toggle the layout between 'grid' and 'stack'
    setLayout((prevLayout) => (prevLayout === "grid" ? "stack" : "grid"));
    
    // Wait for React to re-render the layout change
    await new Promise((resolve) => setTimeout(resolve, 0));

    // Animate the flip
    Flip.from(state, {
      duration: 0.6,
      absolute: true,
      scale: true,
      stagger: -0.1,
      spin: true,
      ease: "power1.easeOut",
    });
  };

  useEffect(() => {
    // Add a click event listener to flip layout on window click
    window.addEventListener("click", flipLayout);

    // Cleanup event listener on unmount
    return () => {
      window.removeEventListener("click", flipLayout);
    };
  }, []);

  return (
    <div className="relative flex items-center justify-center h-screen cursor-pointer">
      <h1 className="absolute text-lg capitalize transform -translate-x-1/2 top-1/4 left-1/2">
        Click anywhere to FLIP
      </h1>
      <div className={layout === "grid" ? "grid grid-cols-5 gap-2" : "flex"}>
        {[...Array(10)].map((_, id) => (
          <Image
          width={24}
          height={24}
            key={id}
            className={`circle w-24 h-24 border-4 border-gray-800 ${
              layout === "grid" ? "rounded-lg" : "rounded-full -ml-10"
            }`}
            src={`/image/sponsorts/music/AuBDM.png`}
            alt="Placeholder"
          />
        ))}
      </div>
    </div>
  );
};

export default FlipLayout;
